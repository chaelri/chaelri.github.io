Step 1 · Concept

How the signal travels

The Beacon shouts a short packet ten times a second. The Finder hears it, measures how loud it was, and converts loudness into a distance estimate.

1 · Broadcast
Beacon sends an 8-byte packet every 100 ms. No WiFi network needed — ESP-NOW is direct radio.
2 · Travel
2.4 GHz radio waves carry the packet. Walls + bodies absorb energy — the further, the quieter.
3 · Capture
Finder reads the RSSI from the radio's rx-control header. Higher (closer to 0) = stronger.
4 · Smooth
RSSI jitters even when still. Rolling average of the last 15 readings hides the wobble.
5 · Display
Log-distance model converts smoothed dBm → metres. Pushed to the 72×40 OLED with a hot/cold tag.
Step 2 · Hardware

Two boards, hand-drawn

Both are ESP32-C3 Super Minis. Only difference: the Finder has a 0.42″ OLED soldered on. No external components — battery + USB-C is enough.

Step 4 · The math

From dBm to metres

RSSI (received signal strength indicator) is measured in dBm — a logarithmic unit. Closer to 0 = stronger signal. We map it to distance using the log-distance path loss model.

The formula
distance = 10((MP − RSSI) / (10 · N))
MP
Measured Power — the RSSI you got when the boards were exactly 1 m apart. Default -50 dBm.
N
Path-loss exponent — how quickly signal fades in your environment. Default 2.5.
RSSI
The smoothed RSSI value the Finder just measured.
Pick N for your space
2.0
Free space
Outdoor, line of sight, no obstacles
2.5
Indoor (default)
Single floor, light walls, furniture
3.0
Heavy indoor
Multiple walls, doors, partial concrete
4.0+
Hard environment
Concrete, metal, multiple floors
warning
Realistic accuracy expectations
  • • Line of sight, calibrated: ±1–2 m up to ~20 m
  • • Through one wall: ±3–5 m (your N is now wrong by ~0.5)
  • • Through multiple walls / floors: direction-only — distance number is fiction, but "closer / further" still works
  • • Hand placement: covering the antenna with your palm easily moves RSSI by 10 dB. Hold the Finder by the USB-C end.

// Seeker · BEACON  (Board A — no OLED)
// Broadcasts ESP-NOW packets at 10 Hz. LED slow-blinks alone, fast-blinks when Finder is hearing.
// REQUIRES Arduino-ESP32 core v3.x (recv-callback signature changed in v3).

#include <WiFi.h>
#include <esp_now.h>
#include <esp_wifi.h>

// ===== EDIT THIS: Finder's MAC (from mac_printer.ino) =====
uint8_t FINDER_MAC[6] = { 0xAA, 0xBB, 0xCC, 0xDD, 0xEE, 0xFF };

const int      LED_PIN              = 8;
const uint32_t BROADCAST_INTERVAL   = 100;   // ms (10 Hz)
const uint32_t ACK_TIMEOUT_MS       = 3000;
const uint32_t BLINK_FAST_PERIOD_MS = 250;
const uint32_t BLINK_SLOW_PERIOD_MS = 1000;

struct __attribute__((packed)) BeaconPacket {
  uint32_t seq;
  uint32_t ms;
};

BeaconPacket pkt   = {0, 0};
uint32_t lastSendMs = 0;
uint32_t lastAckMs  = 0;
bool     finderHeard = false;

void onDataSent(const uint8_t *mac, esp_now_send_status_t status) { }

// v3.x recv-callback signature — `info` carries src_addr + rx_ctrl.
void onDataRecv(const esp_now_recv_info_t *info, const uint8_t *data, int len) {
  lastAckMs    = millis();
  finderHeard  = true;
}

void setup() {
  pinMode(LED_PIN, OUTPUT);
  digitalWrite(LED_PIN, HIGH);
  Serial.begin(115200);
  delay(500);

  WiFi.mode(WIFI_STA);
  WiFi.disconnect();

  if (esp_now_init() != ESP_OK) {
    Serial.println("ESP-NOW init failed");
    while (true) delay(1000);
  }
  esp_now_register_send_cb(onDataSent);
  esp_now_register_recv_cb(onDataRecv);

  esp_now_peer_info_t peer = {};
  memcpy(peer.peer_addr, FINDER_MAC, 6);
  peer.channel = 0; peer.encrypt = false;
  esp_now_add_peer(&peer);

  Serial.printf("beacon ready · my MAC: %s\n", WiFi.macAddress().c_str());
}

void loop() {
  uint32_t now = millis();

  if (now - lastSendMs >= BROADCAST_INTERVAL) {
    lastSendMs = now;
    pkt.seq++;
    pkt.ms = now;
    esp_now_send(FINDER_MAC, (uint8_t *)&pkt, sizeof(pkt));
  }

  if (finderHeard && (now - lastAckMs > ACK_TIMEOUT_MS)) finderHeard = false;

  uint32_t period = finderHeard ? BLINK_FAST_PERIOD_MS : BLINK_SLOW_PERIOD_MS;
  bool ledOn = ((now / (period / 2)) % 2) == 0;
  digitalWrite(LED_PIN, ledOn ? LOW : HIGH);   // active LOW
}




// Seeker · FINDER  (Board B — has the 0.42" OLED)
// Receives ESP-NOW, smooths RSSI, draws distance on the 72×40 OLED.
// REQUIRES: Arduino-ESP32 core v3.x  +  U8g2 library.

#include <WiFi.h>
#include <esp_now.h>
#include <esp_wifi.h>
#include <Wire.h>
#include <U8g2lib.h>
#include <Preferences.h>

// ===== EDIT THIS: Beacon's MAC =====
uint8_t BEACON_MAC[6] = { 0x11, 0x22, 0x33, 0x44, 0x55, 0x66 };

// ===== Tuning =====
float MEASURED_POWER = -50.0f;
float N_FACTOR       =  2.5f;

const int LED_PIN = 8;
const int BTN_PIN = 9;     // BOOT
const int SDA_PIN = 5;     // try 6 if OLED is blank
const int SCL_PIN = 6;     // try 7 if OLED is blank

U8G2_SSD1306_72X40_ER_F_HW_I2C oled(U8G2_R0, U8X8_PIN_NONE);

const int   SAMPLES        = 15;
const uint32_t SIGNAL_TIMEOUT = 1500;
int rssiBuf[SAMPLES] = {0};
int rssiIdx = 0, rssiCount = 0;
uint32_t lastPacketMs = 0;

// button
const uint32_t LONG_PRESS_MS = 1500;
bool     btnLast = HIGH;
uint32_t btnDownMs = 0;
bool     longFired = false;

// calibration
const uint32_t CALIB_DURATION_MS = 5000;
bool     calibActive = false;
uint32_t calibStartMs = 0;
long     calibSum = 0;
int      calibCount = 0;

Preferences prefs;

void onDataRecv(const esp_now_recv_info_t *info, const uint8_t *data, int len) {
  int rssi = info->rx_ctrl->rssi;
  lastPacketMs = millis();

  rssiBuf[rssiIdx] = rssi;
  rssiIdx = (rssiIdx + 1) % SAMPLES;
  if (rssiCount < SAMPLES) rssiCount++;

  if (calibActive) { calibSum += rssi; calibCount++; }

  uint8_t ack = 0x01;
  esp_now_send(info->src_addr, &ack, 1);   // tell beacon we hear it
}

float averageRssi() {
  if (rssiCount == 0) return 0;
  long s = 0;
  for (int i = 0; i < rssiCount; i++) s += rssiBuf[i];
  return (float)s / rssiCount;
}

float rssiToDistance(float rssi) {
  return powf(10.0f, (MEASURED_POWER - rssi) / (10.0f * N_FACTOR));
}

const char *hotColdLabel(float d) {
  if (d < 0.5f)  return "BURNING";
  if (d < 1.5f)  return "HOT";
  if (d < 4.0f)  return "WARM";
  if (d < 10.0f) return "COOL";
  return "COLD";
}
