01 · Overview

How a click travels

Phone tap → cloud database → ESP32 polls → servo arm bursts onto the button → bursts back to rest.

touch_app Trigger
Phone web app writes a string ("click") to one Firebase path.
sync Poll
ESP32 reads that path every 1s via REST. When non-empty, it acts and clears.
bolt Actuate
GPIO3 → ESP32Servo PWM. Toggle: 1300 µs burst engages, motor stops, arm latched. Tap again → mirrored 1700 µs burst back to rest. Single click: same press, ~150 ms hold, auto-return.

03 · Wiring

All four connections

Three jumpers — signal, 5V, GND — straight from ESP32 to the servo's three leads. No relay, no MOSFET, no flyback diode. Powerbank powers both ESP32 and servo on the same rail.

# Connection From To Wire Method
warning
Polarity matters
Servo wire colors are the rule: red → ESP32 5V, brown → GND, orange/yellow → GPIO3 (signal wire is orange on most MG90S, yellow on some clones). Swapping red and brown can release the magic smoke.
info
Tuning the press
CR servos rarely stop at exactly 1500 µs — first nudge STOP_US (1480..1520) until the horn truly idles. Then tune PUSH_MS / RETURN_MS together for travel (longer = deeper press). If the arm spins the wrong way on press, swap PUSH_US and RETURN_US.

#include <WiFi.h>
#include <WiFiMulti.h>
#include <HTTPClient.h>
#include <WebServer.h>
#include <ESP32Servo.h>

// Three trigger paths:
//   1. Phone remote (online)    → writes "click" to Firebase, ESP32 polls
//   2. Local web UI (online)    → http://<esp32-ip>/ on the same WiFi
//   3. SoftAP fallback (offline) → broadcasts "AutoClicker-AP" if no WiFi
WiFiMulti wifiMulti;
WebServer server(80);
Servo finger;
bool inSoftAP = false;

const char* AP_SSID = "AutoClicker-AP";
const char* AP_PASS = "click1234";
const unsigned long WIFI_CONNECT_TIMEOUT_MS = 15000;

const char* DB_URL =
  "https://test-database-55379-default-rtdb.asia-southeast1.firebasedatabase.app"
  "/autoclicker/command.json";

const int SERVO_PIN     = 3;     // GPIO3 → servo signal (orange/yellow)
const int LED_PIN       = 8;     // onboard blue LED (active LOW)

// 360° continuous-rotation servo: pulse width = speed + direction (not angle)
const int STOP_US       = 1500;  // neutral — tune 1480..1520 if the horn creeps
const int PUSH_US       = 1300;  // press direction (swap with RETURN_US if reversed)
const int RETURN_US     = 1700;  // release direction
const int PUSH_MS       = 450;   // burst onto button — longer = more travel/torque
const int RETURN_MS     = 450;   // MUST equal PUSH_MS so release lands at rest
const int CLICK_HOLD_MS = 150;   // momentary-click hold time
const int POLL_MS       = 1000;

bool isPressed = false;          // latched state for toggle

// Inline HTML/CSS for the built-in click UI — no CDN deps so it works offline
// in SoftAP mode. ~150 lines of dark-slate + amber-glow custom CSS, omitted
// here for brevity. See autoclicker/firmware/autoclicker.ino on GitHub for
// the full PROGMEM literal.
const char INDEX_HTML[] PROGMEM = R"rawliteral(
  <!-- dark-slate body, big amber CLICK button with halo + pulse-ring,
       JetBrains-Mono status pill, fetch('/click'). Fully offline-capable. -->
)rawliteral";

void doPress() {
  if (isPressed) return;
  digitalWrite(LED_PIN, LOW);
  finger.writeMicroseconds(PUSH_US);    // burst onto button
  delay(PUSH_MS);
  finger.writeMicroseconds(STOP_US);    // motor off — arm stays put
  isPressed = true;
}

void doRelease() {
  if (!isPressed) return;
  finger.writeMicroseconds(RETURN_US);  // mirrored burst back
  delay(RETURN_MS);
  finger.writeMicroseconds(STOP_US);    // motor off at rest
  digitalWrite(LED_PIN, HIGH);
  isPressed = false;
}

void doToggle() { isPressed ? doRelease() : doPress(); }

void doClick() {
  // Momentary click: always ends back at rest
  if (isPressed) doRelease();
  doPress();
  delay(CLICK_HOLD_MS);
  doRelease();
}

void sendState() {
  String body = String("{\"pressed\":") + (isPressed ? "true" : "false") + "}";
  server.send(200, "application/json", body);
}

void handleRoot()    { server.send_P(200, "text/html; charset=utf-8", INDEX_HTML); }
void handlePress()   { doPress();   sendState(); }
void handleRelease() { doRelease(); sendState(); }
void handleToggle()  { doToggle();  sendState(); }
void handleClick()   { doClick();   sendState(); }
void handleStateGet(){ sendState(); }

void clearCommand() {
  HTTPClient http;
  http.begin(DB_URL);
  http.addHeader("Content-Type", "application/json");
  http.PUT("\"\"");
  http.end();
}

bool tryStation() {
  WiFi.mode(WIFI_STA);
  unsigned long t0 = millis();
  while (wifiMulti.run() != WL_CONNECTED && millis() - t0 < WIFI_CONNECT_TIMEOUT_MS) {
    delay(300);
  }
  return WiFi.status() == WL_CONNECTED;
}

void startSoftAP() {
  WiFi.mode(WIFI_AP);
  WiFi.softAP(AP_SSID, AP_PASS);
  inSoftAP = true;
  Serial.println("SoftAP IP: " + WiFi.softAPIP().toString());
}

void setup() {
  Serial.begin(115200);

  finger.setPeriodHertz(50);
  finger.attach(SERVO_PIN, 500, 2400);
  finger.writeMicroseconds(STOP_US);    // motor stopped at boot

  pinMode(LED_PIN, OUTPUT);
  digitalWrite(LED_PIN, HIGH);

  wifiMulti.addAP("CAYNO", "lokomoko");
  wifiMulti.addAP("Charlie's iPhone", "charlie24");

  if (!tryStation()) startSoftAP();

  server.on("/", handleRoot);
  server.on("/press",   HTTP_POST, handlePress);
  server.on("/release", HTTP_POST, handleRelease);
  server.on("/toggle",  HTTP_POST, handleToggle);
  server.on("/click",   HTTP_POST, handleClick);
  server.on("/state",   HTTP_GET,  handleStateGet);
  server.onNotFound(handleRoot);
  server.begin();
}

void loop() {
  server.handleClient();

  // Firebase polling only when connected to WiFi (not in SoftAP)
  static unsigned long lastPoll = 0;
  if (!inSoftAP && millis() - lastPoll >= POLL_MS) {
    lastPoll = millis();
    if (wifiMulti.run() != WL_CONNECTED) return;

    HTTPClient http;
    http.begin(DB_URL);
    if (http.GET() == 200) {
      String body = http.getString();
      body.replace("\"", ""); body.trim();
      if      (body == "press")    { doPress();   clearCommand(); }
      else if (body == "release")  { doRelease(); clearCommand(); }
      else if (body == "toggle")   { doToggle();  clearCommand(); }
      else if (body == "click")    { doClick();   clearCommand(); }
    }
    http.end();
  }
  delay(2);
}
