01 · Overview

How a click travels

Phone tap → cloud database → ESP32 polls → relay clicks → solenoid pushes the button physically.

touch_app Trigger
Phone web app writes a string ("click") to one Firebase path.
sync Poll
ESP32 reads that path every 1s via REST. When non-empty, it acts and clears.
bolt Actuate
GPIO3 → relay coil → COM/NO closes → solenoid fires. 200ms HIGH = one button press.

03 · Wiring

All seven connections

Three signal-side jumpers, the solenoid permanently at +5V on the relay's VCC pin shaft, and a bridge wire that lets the relay open/close the GND return path. Low-side switching — no soldering.

Polarity check
5V → VCC and GND → GND must not be swapped. The header pin order on the relay module is silkscreened — read it before plugging.

#include <WiFi.h>
#include <WiFiMulti.h>
#include <HTTPClient.h>
#include <WebServer.h>

// Three trigger paths:
//   1. Phone remote (online)    → writes "click" to Firebase, ESP32 polls
//   2. Local web UI (online)    → http://<esp32-ip>/ on the same WiFi
//   3. SoftAP fallback (offline) → broadcasts "AutoClicker-AP" if no WiFi
WiFiMulti wifiMulti;
WebServer server(80);
bool inSoftAP = false;

const char* AP_SSID = "AutoClicker-AP";
const char* AP_PASS = "click1234";
const unsigned long WIFI_CONNECT_TIMEOUT_MS = 15000;

const char* DB_URL =
  "https://test-database-55379-default-rtdb.asia-southeast1.firebasedatabase.app"
  "/autoclicker/command.json";

const int RELAY_PIN = 3;
const int LED_PIN   = 8;
const int CLICK_MS  = 200;
const int POLL_MS   = 1000;

const bool ACTIVE_LOW = false;
inline int relayOn()  { return ACTIVE_LOW ? LOW  : HIGH; }
inline int relayOff() { return ACTIVE_LOW ? HIGH : LOW;  }

// Inline HTML/CSS for the built-in click UI — no CDN deps so it works offline
// in SoftAP mode. ~150 lines of dark-slate + amber-glow custom CSS, omitted
// here for brevity. See autoclicker/firmware/autoclicker.ino on GitHub for
// the full PROGMEM literal.
const char INDEX_HTML[] PROGMEM = R"rawliteral(
  <!-- dark-slate body, big amber CLICK button with halo + pulse-ring,
       JetBrains-Mono status pill, fetch('/click'). Fully offline-capable. -->
)rawliteral";

void click(int times) {
  for (int i = 0; i < times; i++) {
    digitalWrite(RELAY_PIN, relayOn());
    digitalWrite(LED_PIN, LOW);
    delay(CLICK_MS);
    digitalWrite(RELAY_PIN, relayOff());
    digitalWrite(LED_PIN, HIGH);
    if (i < times - 1) delay(150);
  }
}

void handleRoot()  { server.send_P(200, "text/html; charset=utf-8", INDEX_HTML); }
void handleClick() { click(1); server.send(200, "application/json", "{\"ok\":true}"); }

void clearCommand() {
  HTTPClient http;
  http.begin(DB_URL);
  http.addHeader("Content-Type", "application/json");
  http.PUT("\"\"");
  http.end();
}

bool tryStation() {
  WiFi.mode(WIFI_STA);
  unsigned long t0 = millis();
  while (wifiMulti.run() != WL_CONNECTED && millis() - t0 < WIFI_CONNECT_TIMEOUT_MS) {
    delay(300);
  }
  return WiFi.status() == WL_CONNECTED;
}

void startSoftAP() {
  WiFi.mode(WIFI_AP);
  WiFi.softAP(AP_SSID, AP_PASS);
  inSoftAP = true;
  Serial.println("SoftAP IP: " + WiFi.softAPIP().toString());
}

void setup() {
  Serial.begin(115200);
  pinMode(RELAY_PIN, OUTPUT);
  pinMode(LED_PIN, OUTPUT);
  digitalWrite(RELAY_PIN, relayOff());
  digitalWrite(LED_PIN, HIGH);

  wifiMulti.addAP("CAYNO", "lokomoko");
  wifiMulti.addAP("Charlie's iPhone", "charlie24");

  if (!tryStation()) startSoftAP();

  server.on("/", handleRoot);
  server.on("/click", HTTP_POST, handleClick);
  server.on("/click", HTTP_GET, handleClick);
  server.onNotFound(handleRoot);
  server.begin();
}

void loop() {
  server.handleClient();

  // Firebase polling only when connected to WiFi (not in SoftAP)
  static unsigned long lastPoll = 0;
  if (!inSoftAP && millis() - lastPoll >= POLL_MS) {
    lastPoll = millis();
    if (wifiMulti.run() != WL_CONNECTED) return;

    HTTPClient http;
    http.begin(DB_URL);
    if (http.GET() == 200) {
      String body = http.getString();
      body.replace("\"", ""); body.trim();
      if (body == "click")           { click(1); clearCommand(); }
      else if (body == "double")     { click(2); clearCommand(); }
      else if (body.startsWith("auto_")) {
        int n = body.substring(5).toInt();
        if (n > 0 && n <= 50) { click(n); clearCommand(); }
      }
    }
    http.end();
  }
  delay(2);
}
