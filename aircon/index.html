01 · Overview

How a command travels

Phone tap → cloud database → ESP32 polls → IR LED flashes the TCL112AC pattern → aircon obeys.

touch_app Trigger
Phone web app writes a state object ({power, mode, temp, fan}) to one Firebase path.
sync Poll
ESP32 reads that path every 1s via REST. When non-empty, it parses the state, sends, and clears.
flare Transmit
GPIO3 pulses the IR LED at 38 kHz in the TCL112AC pattern (~150 ms · 112 bits). Aircon decodes and obeys.

03 · Wiring

All three connections

Two wires from ESP32 to the LED (signal via 100 Ω resistor + ground), one USB-C to the wall charger. No transistor, no heatsink, no batteries.

# Connection From To Wire Method
warning
Polarity matters
IR LEDs are polar — long leg = anode (+) goes through the 100 Ω resistor to GPIO3; short leg = cathode (−) goes to GND. If you wire it backwards the LED just won't light (no damage), but the aircon won't respond either. Look at the bulb: the lead with the smaller chip-anvil inside is the anode.
info
Why the resistor
A bare IR LED across 3.3 V would draw past the GPIO's 40 mA per-pin limit — risks slowly cooking the pin or burning the LED. The 100 Ω resistor caps current to safe levels at 38 kHz / ~33 % duty cycle. Brown–Black–Brown–Gold bands.

#include <WiFi.h>
#include <WiFiMulti.h>
#include <HTTPClient.h>
#include <WebServer.h>
#include <IRremoteESP8266.h>
#include <IRsend.h>
#include <ir_Tcl.h>

// Three trigger paths:
//   1. Phone remote (online)    → writes JSON state to Firebase, ESP32 polls
//   2. Local web UI (online)    → http://<esp32-ip>/ on the same WiFi
//   3. SoftAP fallback (offline) → broadcasts "Aircon-AP" if no WiFi
WiFiMulti wifiMulti;
WebServer server(80);
bool inSoftAP = false;

const char* AP_SSID = "Aircon-AP";
const char* AP_PASS = "aircon24";
const unsigned long WIFI_CONNECT_TIMEOUT_MS = 15000;

const char* DB_URL =
  "https://test-database-55379-default-rtdb.asia-southeast1.firebasedatabase.app"
  "/aircon/command.json";
const char* STATE_URL =
  "https://test-database-55379-default-rtdb.asia-southeast1.firebasedatabase.app"
  "/aircon/state.json";

const int IR_LED_PIN = 3;     // GPIO3 → 100Ω resistor → IR LED anode
const int LED_PIN    = 8;     // onboard blue LED — blinks on send
const int POLL_MS    = 1000;

IRTcl112Ac ac(IR_LED_PIN);

struct AcState {
  bool   power = false;
  uint8_t mode = kTcl112AcCool;
  float   temp = 24.0f;
  uint8_t fan  = kTcl112AcFanAuto;
  bool   swing = false;
} state;

void sendIR() {
  digitalWrite(LED_PIN, LOW);
  ac.setPower(state.power);
  ac.setMode(state.mode);
  ac.setTemp(state.temp);
  ac.setFan(state.fan);
  ac.setSwingVertical(state.swing);
  ac.send();
  digitalWrite(LED_PIN, HIGH);
}

// Naive JSON parser — only the fields we care about. Avoids ArduinoJson dep.
String extractStr(const String& body, const char* key) {
  String pat = String("\"") + key + "\":\"";
  int i = body.indexOf(pat);
  if (i < 0) return "";
  i += pat.length();
  int j = body.indexOf('"', i);
  return j < 0 ? "" : body.substring(i, j);
}

void applyCommand(const String& body) {
  String p = extractStr(body, "power");
  if (p == "on")       state.power = true;
  else if (p == "off") state.power = false;

  String m = extractStr(body, "mode");
  if      (m == "cool") state.mode = kTcl112AcCool;
  else if (m == "dry")  state.mode = kTcl112AcDry;
  else if (m == "fan")  state.mode = kTcl112AcFan;
  else if (m == "heat") state.mode = kTcl112AcHeat;
  else if (m == "auto") state.mode = kTcl112AcAuto;

  // ... temp / fan / swing handled the same way

  sendIR();
  publishState();
}

void setup() {
  Serial.begin(115200);
  ac.begin();
  pinMode(LED_PIN, OUTPUT);
  digitalWrite(LED_PIN, HIGH);

  wifiMulti.addAP("CAYNO", "lokomoko");
  wifiMulti.addAP("Chaelri", "charlie24");

  if (!tryStation()) startSoftAP();

  server.on("/", handleRoot);
  server.on("/state", HTTP_GET,  handleState);
  server.on("/set",   HTTP_POST, handleSet);
  server.begin();
  publishState();
}

void loop() {
  server.handleClient();

  static unsigned long lastPoll = 0;
  if (!inSoftAP && millis() - lastPoll >= POLL_MS) {
    lastPoll = millis();
    if (wifiMulti.run() != WL_CONNECTED) return;

    HTTPClient http;
    http.begin(DB_URL);
    if (http.GET() == 200) {
      String body = http.getString(); body.trim();
      if (body.length() > 2 && body != "null" && body != "\"\"") {
        applyCommand(body);
        clearCommand();
      }
    }
    http.end();
  }
  delay(2);
}
