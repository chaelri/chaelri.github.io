Overview

Where each tap goes

Three targets, three Firebase paths. Sibling toggles (lights, aircon) write the same payload the phone remotes use, so the existing devices can't tell the difference. Mood pings drop into /tayo/moods/charlie and surface in the Tayo PWA feed.

1. Tap on any screen
B3/B4 always work — sibling toggles aren't gated by what screen you're on.
2. HTTPS PUT
One REST write to Firebase. Lights & aircon use the same payload as the phone remotes.
3. Target reacts
Servo / IR / Moods Feed responds within ~150–1000 ms.
Hardware

One dev board, four buttons, a powerbank

ESP32-C3 SuperMini with the 0.42" OLED soldered on — same I²C pinout (GPIO5/6) and onboard blue LED (GPIO8) as the autoclicker. Four 6×6×5 mm tactile buttons hang off GPIO 0/1/3/4 to GND with the internal pull-ups handling the resistor side. Power is a USB-C mini powerbank plugged straight into the ESP32's own USB-C port — no TP4056, no LiPo, no boost converter, no transistor. The diagram below also shows the optional LiPo + TP4056 stack from pocket-remote if you want untethered runtime.

ESP32-C3 SuperMini + OLED
0.42" SSD1306 on GPIO5/6. Blue LED on GPIO8 (active-LOW). Avoid GPIO9 (BOOT strap).
4× tactile button
6×6×5 mm momentary. Each leg: GPIO ↔ GND. INPUT_PULLUP handles the pull-up.
USB-C mini powerbank
Any 5 V powerbank with USB-C out → ESP32-C3's USB-C in. Done.
Optional: LiPo + TP4056
If you want a self-contained build later, mirror pocket-remote's stack.
Wiring

Four button legs + a USB-C cable

Eight short wires land the tactile buttons, plus one USB-C cable from a mini powerbank into the ESP32's own port. Every button has the same pattern: one leg to its GPIO, the other leg to the shared GND rail (the ESP32's GD pad fans out). No external resistors — INPUT_PULLUP in firmware does the job.

# From To Wire / part Notes
bolt Powering it

Plug any USB-C mini powerbank into the ESP32-C3's own USB-C port — 5 V goes through the onboard LDO and the OLED + radio run happily. The button pins GPIO 0, 1, 3 and 4 are all clean general-purpose I/O on the C3 (the strapping pins are GPIO 2, 8 and 9 — we deliberately avoid them or only drive 8 as an output).

Demo

OLED preview

Animation only — no Firebase calls. Use the simulated buttons to flip between the three screens, run the 5-minute timer (sped up ×60 for the preview), or cycle through mood faces. B3/B4 fire from any screen.

Tip: the timer runs ×60 here so you don't have to wait 5 minutes for the screen to flash. On the real device, ticks are 1 second.

Four-button scan with tap / hold dispatch
Same shape per button — debounce + tap-on-release / hold-on-cross

// All four buttons share a struct so we never duplicate the state machine.
// B1/B2 emit both tap and hold events; B3/B4 only need tap.
struct Btn {
  uint8_t       pin;
  int           lastLevel;
  bool          pressed;
  bool          holdFired;
  unsigned long lastEdge;
  unsigned long downAt;
};

Btn B[4] = {
  { 0, HIGH, false, false, 0, 0 },   // B1 MODE
  { 1, HIGH, false, false, 0, 0 },   // B2 ACTION
  { 3, HIGH, false, false, 0, 0 },   // B3 LIGHTS
  { 4, HIGH, false, false, 0, 0 },   // B4 AIRCON
};

const unsigned long DEBOUNCE_MS = 25;
const unsigned long TAP_MAX_MS  = 500;
const unsigned long HOLD_MS     = 800;

void scanButtons() {
  unsigned long now = millis();
  for (int i = 0; i < 4; ++i) {
    Btn& b = B[i];
    int level = digitalRead(b.pin);
    if (level != b.lastLevel) { b.lastLevel = level; b.lastEdge = now; continue; }
    if (now - b.lastEdge < DEBOUNCE_MS) continue;

    bool nowPressed = (level == LOW);
    if (nowPressed != b.pressed) {
      b.pressed = nowPressed;
      if (b.pressed) { b.downAt = now; b.holdFired = false; }
      else if (!b.holdFired && (now - b.downAt) <= TAP_MAX_MS) dispatchTap(i);
    } else if (b.pressed && !b.holdFired && (now - b.downAt) >= HOLD_MS) {
      b.holdFired = true;
      dispatchHold(i);
    }
  }
}

B1 cycles. B2 acts within the current screen. B3/B4 are screen-agnostic.

enum Screen { CLOCK, TIMER, MOOD, SCREEN_COUNT };
Screen current = CLOCK;

void dispatchTap(int i) {
  switch (i) {
    case 0: current = (Screen)((current + 1) % SCREEN_COUNT); saveScreen(); break;
    case 1: actionTap();   break;   // context-dependent
    case 2: fireLights();  break;   // any screen
    case 3: fireAircon();  break;   // any screen
  }
}

void actionTap() {
  switch (current) {
    case CLOCK: break;                          // no-op on the clock
    case TIMER: timerToggle(); break;            // start ↔ pause
    case MOOD:  moodIdx = (moodIdx + 1) % MOOD_COUNT; break;
  }
}

void dispatchHold(int i) {
  if (i != 1) return;   // only B2 has hold
  switch (current) {
    case CLOCK: break;
    case TIMER: timerReset(); break;
    case MOOD:  sendMood(moodIdx); break;
  }
}

unsigned long timerEndMs = 0;   // 0 = idle
bool          timerDone  = false;

void timerToggle() {
  if (timerDone) { timerReset(); return; }
  if (timerEndMs == 0) timerEndMs = millis() + 5 * 60 * 1000UL;
  else                   timerEndMs = 0;   // pause -> back to idle (5:00)
}
void timerReset() { timerEndMs = 0; timerDone = false; }

void timerTick() {
  if (timerEndMs && millis() >= timerEndMs) {
    timerEndMs = 0;
    timerDone  = true;
  }
  if (timerDone) {
    bool on = ((millis() / 280) % 2) == 0;
    digitalWrite(LED_PIN, on ? LOW : HIGH);    // GPIO8 is active-LOW
    oled.setDrawColor(on ? 0 : 1);             // invert pixels each frame
    drawTimerScreen("TIME UP");
  }
}

const char* CLICK_URL = "https://test-database-55379-default-rtdb.asia-southeast1.firebasedatabase.app/autoclicker/command.json";
const char* AC_URL    = "https://test-database-55379-default-rtdb.asia-southeast1.firebasedatabase.app/aircon/command.json";
const char* MOOD_BASE = "https://test-database-55379-default-rtdb.asia-southeast1.firebasedatabase.app/tayo/moods/charlie";

bool putJson(const char* url, const char* body) {
  if (WiFi.status() != WL_CONNECTED) return false;
  HTTPClient http;
  if (!http.begin(url)) return false;
  http.addHeader("Content-Type", "application/json");
  int code = http.PUT(body);
  http.end();
  return (code >= 200 && code < 300);
}

bool fireLights() { return putJson(CLICK_URL, "\"toggle\""); }
bool fireAircon() { return putJson(AC_URL,    "{\"cmd\":\"power\"}"); }

// Mood ping: POST so RTDB auto-generates a push-id child key.
bool sendMood(int idx) {
  const char* name = MOODS[idx].name;
  char body[128];
  snprintf(body, sizeof(body),
    "{\"emotion\":\"%s\",\"ts\":%lu,\"source\":\"pocket-dash\"}",
    name, (unsigned long)time(nullptr));
  char url[200];
  snprintf(url, sizeof(url), "%s.json", MOOD_BASE);
  HTTPClient http; http.begin(url);
  http.addHeader("Content-Type", "application/json");
  int code = http.POST(body);
  http.end();
  return (code >= 200 && code < 300);
}
